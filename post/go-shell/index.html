#!/bin/bash
# 部署到 github pages 脚本
# 错误时终止脚本
set -e

# 删除打包文件夹
/bin/rm -rf public

# 打包。even 是主题
hugo -t even # if using a theme, replace with `hugo -t <YOURTHEME>`

# 进入打包文件夹
cd public

# Add changes to git.

git init
git add -A

# Commit changes.
msg="building site `date`"
if [ $# -eq 1 ]
  then msg="$1"
fi
git commit -m "$msg"

# 推送到githu  
# nusr.github.io 只能使用 master分支
git push -f <youraddress> master

# 回到原文件夹
cd ..

package main

import(
	"fmt"
	"os/exec"
	"os"
	"time"
)

func EE(err error){
	if err != nil{
		fmt.Println(err)
		os.Exit(1)
	}
}
func main(){
	sitedir := "$hugopath"
	fmt.Println("[0]cd ",sitedir)
	err0 := os.Chdir(sitedir)
	EE(err0)

	pdir := "$hogopath/public"
	fmt.Println("[1]rm",pdir)
	cmd1 := exec.Command("/bin/rm","-rf",pdir)
	_, err2 := cmd1.Output()
	EE(err2)
	
	fmt.Println("[2]hugo cmd")
	cmd2 := exec.Command("hugo","-t","even")
	out,err := cmd2.CombinedOutput()
	EE(err)
	fmt.Println(string(out))
	
	// git

	fmt.Println("[3]git")
	err3 := os.Chdir("public")
	EE(err3)

	cmd4 := exec.Command("git","init")
	_,err4 := cmd4.CombinedOutput()
	EE(err4)
	cmd5 := exec.Command("git","add","-A")
	out5,err5 := cmd5.CombinedOutput()
	EE(err5)
	fmt.Println(string(out5))

	msg := "building site at"
	fmt.Println(msg,time.Now())
	cmd6 := exec.Command("git","commit","-m",msg)
	out6,err6 := cmd6.CombinedOutput()
	EE(err6)
	fmt.Println(string(out6))

	fmt.Println("[4] push")
	remote := "https://github.com/yourid/yourid.github.io.git"
	cmd7 := exec.Command("git","push","-f",remote,"master")
	out7,err7 := cmd7.CombinedOutput()
	EE(err7)
	fmt.Println(string(out7))
	
	os.Chdir(sitedir)
	fmt.Println("[5] Done")
}
